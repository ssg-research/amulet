% The wasy font (used only for the \smark/\xmark/\cmark circle glyphs below)
% ships a single medium series at fixed design sizes, so those glyphs in a bold
% caption or at small sizes otherwise trigger font-substitution warnings. We
% declare the family and shapes up front (before any glyph is used, which also
% stops the lazy uwasy.fd from overriding these): medium scales from wasy10 at
% any size, and bold silently falls back to medium. The rendered glyphs are
% unchanged.
\DeclareFontFamily{U}{wasy}{}
\DeclareFontShape{U}{wasy}{m}{n}{<->wasy10}{}
\DeclareFontShape{U}{wasy}{b}{n}{<->ssub*wasy/m/n}{}

% acmart auto-balances the final page's columns with \balance (from the balance
% package). On a short last page that warns "\balance in second column". Make
% the call a no-op for the draft; remove this for camera-ready to restore
% balancing once the last page fills both columns.
\AtBeginDocument{\let\balance\relax}
